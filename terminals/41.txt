function Remove-IfOld($path) {
  if (-not (Test-Path $path)) { return }
  $item = Get-Item $path
  if ($item.LastWriteTime -lt $cutoff) {
    try {
      Remove-Item -LiteralPath $path -Recurse -Force -ErrorAction Stop
      $script:deleted += $path
    } catch {
      $script:errors += \"$path : $($_.Exception.Message)\"
    }

    $script:kept += \"$path (last: $($item.LastWriteTime))\"
  }
}
